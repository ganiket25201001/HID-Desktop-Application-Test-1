import win32api, win32con, win32gui
import wmi, time, threading, logging
from utils.logger import get_logger

log = get_logger("usb_monitor")

class USBMonitor:
    def __init__(self, on_device_inserted):
        self.wmi_client = wmi.WMI()
        self.on_inserted = on_device_inserted
        self.known_devices = set()
        self.blocked = True   # All USB blocked by default

    def block_usb_storage(self):
        """Disable USB storage via registry (requires Admin)"""
        import winreg
        key_path = r"SYSTEM\CurrentControlSet\Services\USBSTOR"
        key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE,
                               key_path, 0, winreg.KEY_SET_VALUE)
        # Start = 4 means disabled, 3 means enabled
        winreg.SetValueEx(key, "Start", 0, winreg.REG_DWORD, 4)
        winreg.CloseKey(key)
        log.info("USB storage BLOCKED via registry")

    def allow_usb_storage(self):
        """Re-enable USB storage after analysis passes"""
        import winreg
        key_path = r"SYSTEM\CurrentControlSet\Services\USBSTOR"
        key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE,
                               key_path, 0, winreg.KEY_SET_VALUE)
        winreg.SetValueEx(key, "Start", 0, winreg.REG_DWORD, 3)
        winreg.CloseKey(key)
        log.info("USB storage RE-ENABLED")

    def get_usb_devices(self):
        """Return list of currently connected USB drives"""
        devices = []
        for disk in self.wmi_client.Win32_DiskDrive():
            if "USB" in disk.InterfaceType:
                for partition in disk.associators(
                        "Win32_DiskDriveToDiskPartition"):
                    for logical in partition.associators(
                            "Win32_LogicalDiskToPartition"):
                        devices.append({
                            "drive": logical.DeviceID,
                            "serial": disk.SerialNumber,
                            "model": disk.Model,
                            "size": disk.Size
                        })
        return devices

    def start_monitoring(self):
        """Poll for new USB devices every 2 seconds"""
        self.block_usb_storage()  # Lock on startup
        log.info("USB Monitor active — waiting for devices...")
        while True:
            current = {d["serial"]: d for d in self.get_usb_devices()}
            new_devices = set(current.keys()) - self.known_devices
            for serial in new_devices:
                log.warning(f"NEW USB DETECTED: {current[serial]['model']}")
                self.known_devices.add(serial)
                threading.Thread(
                    target=self.on_inserted,
                    args=(current[serial],),
                    daemon=True
                ).start()
            time.sleep(2)

AI-Powered Sandbox Analysis

import os, hashlib, math, requests, pefile, time
from utils.config import cfg
from utils.logger import get_logger

log = get_logger("sandbox")

class SandboxEngine:
    def __init__(self):
        self.vt_key = cfg["VIRUSTOTAL_API_KEY"]
        self.RISKY_EXTENSIONS = {
            '.exe', '.dll', '.bat', '.ps1', '.vbs',
            '.js', '.wsf', '.hta', '.cmd', '.scr',
            '.msi', '.jar', '.lnk', '.pif'
        }

    def collect_files(self, drive_letter):
        """Scan drive and collect all files for analysis"""
        files = []
        drive_path = f"{drive_letter}\\"
        for root, _, filenames in os.walk(drive_path):
            for name in filenames:
                full = os.path.join(root, name)
                try:
                    size = os.path.getsize(full)
                    ext  = os.path.splitext(name)[1].lower()
                    files.append({
                        "path": full,
                        "name": name,
                        "size": size,
                        "ext": ext,
                        "risky": ext in self.RISKY_EXTENSIONS
                    })
                except PermissionError:
                    pass
        log.info(f"Collected {len(files)} files from {drive_letter}")
        return files

    def sha256_file(self, filepath):
        h = hashlib.sha256()
        with open(filepath, "rb") as f:
            for chunk in iter(lambda: f.read(8192), b""):
                h.update(chunk)
        return h.hexdigest()

    def entropy(self, data):
        if not data: return 0
        freq = [data.count(bytes([b])) / len(data) for b in range(256)]
        return -sum(p * math.log2(p) for p in freq if p > 0)

    def virustotal_scan(self, sha256):
        """Check file hash against VirusTotal database"""
        headers = {"x-apikey": self.vt_key}
        r = requests.get(
            f"https://www.virustotal.com/api/v3/files/{sha256}",
            headers=headers, timeout=10
        )
        if r.status_code == 200:
            stats = r.json()["data"]["attributes"]["last_analysis_stats"]
            return stats.get("malicious", 0), stats.get("suspicious", 0)
        return 0, 0

    def analyze_file(self, file_info):
        """Full static analysis pipeline for one file"""
        path   = file_info["path"]
        score  = 0
        flags  = []

        try:
            with open(path, "rb") as f:
                data = f.read()
            sha256 = self.sha256_file(path)

            # VirusTotal cloud check
            mal, sus = self.virustotal_scan(sha256)
            if mal > 0:
                score += mal * 10
                flags.append(f"VT: {mal} engines flagged as malicious")
            if sus > 0:
                score += sus * 3
                flags.append(f"VT: {sus} engines flagged as suspicious")

            # Entropy check (packed/encrypted = malware indicator)
            ent = self.entropy(data)
            if ent > 7.2:
                score += 25
                flags.append(f"High entropy: {ent:.2f} (possible packer)")

            # PE file analysis
            if file_info["ext"] in {'.exe', '.dll', '.scr'}:
                try:
                    pe = pefile.PE(path)
                    suspicious_sec = [s for s in pe.sections
                                      if self.entropy(s.get_data()) > 7.0]
                    if suspicious_sec:
                        score += len(suspicious_sec) * 15
                        flags.append(f"{len(suspicious_sec)} high-entropy PE sections")
                except: pass

            # Risky extension bonus
            if file_info["risky"]:
                score += 10
                flags.append(f"Risky file type: {file_info['ext']}")

        except Exception as e:
            log.error(f"Analysis error on {path}: {e}")

        return {
            "file": file_info,
            "score": min(score, 100),
            "flags": flags,
            "sha256": sha256 if "sha256" in dir() else ""
        }

    def run_full_scan(self, drive_letter):
        """Analyze entire USB drive"""
        files = self.collect_files(drive_letter)
        results = []
        for fi in files:
            log.info(f"Scanning: {fi['name']}")
            results.append(self.analyze_file(fi))
            time.sleep(0.5)  # Respect VT rate limit (4 req/min free)
        return results

Threat Classification Engine

from utils.config import cfg
from utils.logger import get_logger

log = get_logger("classifier")

THRESHOLD_DANGER     = int(cfg.get("SCORE_THRESHOLD_DANGER", 70))
THRESHOLD_SUSPICIOUS = int(cfg.get("SCORE_THRESHOLD_SUSPICIOUS", 35))

def classify_file(result):
    score = result["score"]
    if score >= THRESHOLD_DANGER:
        verdict = "DANGEROUS"
        color   = "#e74c3c"
        action  = "BLOCK"
        icon    = "🔴"
    elif score >= THRESHOLD_SUSPICIOUS:
        verdict = "SUSPICIOUS"
        color   = "#f39c12"
        action  = "WARN"
        icon    = "🟡"
    else:
        verdict = "SAFE"
        color   = "#2ecc71"
        action  = "ALLOW"
        icon    = "🟢"

    result["verdict"] = verdict
    result["color"]   = color
    result["action"]  = action
    result["icon"]    = icon
    log.info(f"{icon} {result['file']['name']}: {verdict} (score {score})")
    return result

def classify_drive(results):
    """Classify all files and produce overall drive verdict"""
    classified = [classify_file(r) for r in results]
    verdicts   = [r["verdict"] for r in classified]
    counts     = {
        "SAFE":      verdicts.count("SAFE"),
        "SUSPICIOUS": verdicts.count("SUSPICIOUS"),
        "DANGEROUS": verdicts.count("DANGEROUS")
    }
    overall = ("DANGEROUS" if counts["DANGEROUS"] > 0
               else "SUSPICIOUS" if counts["SUSPICIOUS"] > 0
               else "SAFE")
    max_score = max((r["score"] for r in classified), default=0)
    return classified, {
        "overall":   overall,
        "max_score": max_score,
        "counts":    counts,
        "total":     len(classified)
    }

import hashlib, customtkinter as ctk
from tkinter import simpledialog, messagebox
from utils.config import cfg
from utils.logger import get_logger

log = get_logger("key_verify")

def hash_key(plain_text):
    return hashlib.sha256(plain_text.encode()).hexdigest()

def setup_security_key(plain_key):
    """Run once during setup to store hashed key in .env"""
    hashed = hash_key(plain_key)
    with open(".env", "a") as f:
        f.write(f"\nSECURITY_KEY_HASH={hashed}\n")
    log.info("Security key saved (hashed). Never stored in plain text.")
    return hashed

def verify_key_gui():
    """Prompt user for security key via dialog, return True if correct"""
    root = ctk.CTk()
    root.withdraw()

    entered = simpledialog.askstring(
        "🔐 Security Verification",
        "Enter your security key to confirm this action:",
        show="*",
        parent=root
    )
    root.destroy()

    if not entered:
        log.warning("Key verification cancelled by user")
        return False

    stored_hash = cfg.get("SECURITY_KEY_HASH", "")
    entered_hash = hash_key(entered)

    if entered_hash == stored_hash:
        log.info("Security key: VERIFIED ✓")
        return True
    else:
        log.warning("Security key: FAILED ✗")
        messagebox.showerror("Verification Failed",
                              "Incorrect security key. Access denied.")
        return False

import os, shutil, subprocess, json, datetime
import winreg
from utils.config import cfg
from utils.logger import get_logger

log = get_logger("access_ctrl")
QUARANTINE = cfg["QUARANTINE_DIR"]
REPORT_DIR = cfg["REPORT_DIR"]

def _enable_usb():
    key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE,
        r"SYSTEM\CurrentControlSet\Services\USBSTOR",
        0, winreg.KEY_SET_VALUE)
    winreg.SetValueEx(key, "Start", 0, winreg.REG_DWORD, 3)
    winreg.CloseKey(key)

def _disable_usb():
    key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE,
        r"SYSTEM\CurrentControlSet\Services\USBSTOR",
        0, winreg.KEY_SET_VALUE)
    winreg.SetValueEx(key, "Start", 0, winreg.REG_DWORD, 4)
    winreg.CloseKey(key)

def apply_decision(device_info, decision, classified_files, summary):
    drive = device_info["drive"]  # e.g. "E:"

    if decision == "ALLOW":
        log.info(f"ALLOW: Enabling USB access for {drive}")
        _enable_usb()
        # Refresh drive so Windows mounts it
        subprocess.run(["powershell", "-Command",
            f"(New-Object -ComObject Shell.Application).NameSpace('{drive}\\').Self.Name"],
            capture_output=True)
        log.info(f"✅ USB {drive} allowed with full access")

    elif decision == "MANAGE":
        log.warning(f"MANAGE: Enabling USB but quarantining threats")
        _enable_usb()
        os.makedirs(QUARANTINE, exist_ok=True)
        for r in classified_files:
            if r["verdict"] != "SAFE":
                src  = r["file"]["path"]
                dest = os.path.join(QUARANTINE, r["file"]["name"])
                try:
                    shutil.copy2(src, dest)
                    os.remove(src)  # Remove from USB
                    log.warning(f"Quarantined: {src}")
                except Exception as e:
                    log.error(f"Quarantine failed for {src}: {e}")
        log.info(f"⚠️ USB {drive} accessible — threats quarantined")

    elif decision == "BLOCK":
        log.critical(f"BLOCK: Keeping USB storage disabled for {drive}")
        _disable_usb()
        # Eject the device safely
        subprocess.run(["powershell", "-Command",
            f"$disk = Get-WmiObject -Query 'SELECT * FROM Win32_LogicalDisk WHERE DeviceID=\"{drive}\"';" +
            "$disk.Dismount()"], capture_output=True)
        log.critical(f"🚫 USB {drive} BLOCKED and ejected")

    # Save JSON report
    _save_report(device_info, decision, classified_files, summary)

def _save_report(device, decision, files, summary):
    os.makedirs(REPORT_DIR, exist_ok=True)
    ts   = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    path = os.path.join(REPORT_DIR, f"usb_report_{ts}.json")
    report = {
        "timestamp": ts,
        "device":    device,
        "decision":  decision,
        "summary":   summary,
        "files": [{"name": r["file"]["name"], "verdict": r["verdict"],
                    "score": r["score"], "flags": r["flags"]} for r in files]
    }
    with open(path, "w") as f:
        json.dump(report, f, indent=2)
    log.info(f"Report saved: {path}")

#!/usr/bin/env python3
# USB Security Gateway — Windows Edition
# Run as Administrator!

import sys, ctypes, logging
from utils.config         import load_config, cfg
from utils.logger         import setup_logger
from core.usb_monitor     import USBMonitor
from core.sandbox_engine  import SandboxEngine
from core.classifier      import classify_drive
from core.decision_gate   import DecisionGate
from core.key_verify      import verify_key_gui
from core.access_controller import apply_decision

def is_admin():
    try: return ctypes.windll.shell32.IsUserAnAdmin()
    except: return False

def on_usb_inserted(device_info):
    log.warning(f"=== USB INSERTED: {device_info['model']} ===")

    # Stage 2: AI Sandbox Analysis
    log.info("[STAGE 2] Running sandbox analysis...")
    engine  = SandboxEngine()
    results = engine.run_full_scan(device_info["drive"])

    # Stage 3: Threat Classification
    log.info("[STAGE 3] Classifying threats...")
    classified, summary = classify_drive(results)

    # Stage 4: User Decision Gate
    log.info("[STAGE 4] Showing decision gate to user...")
    gate     = DecisionGate(summary, classified)
    decision = gate.show()

    if not decision:
        log.warning("No decision made — defaulting to BLOCK")
        decision = "BLOCK"

    # Stage 5: Security Key Verification
    log.info("[STAGE 5] Verifying security key...")
    if not verify_key_gui():
        log.critical("Key verification failed — blocking device")
        decision = "BLOCK"

    # Stage 6: Apply Decision
    log.info(f"[STAGE 6] Applying decision: {decision}")
    apply_decision(device_info, decision, classified, summary)

if __name__ == "__main__":
    if not is_admin():
        # Re-launch as admin
        ctypes.windll.shell32.ShellExecuteW(
            None, "runas", sys.executable, " ".join(sys.argv), None, 1)
        sys.exit()

    load_config()
    log = setup_logger()
    log.info("USB Security Gateway started — ALL USB PORTS LOCKED")

    monitor = USBMonitor(on_device_inserted=on_usb_inserted)
    monitor.start_monitoring()  # Blocks forever, polls every 2s
